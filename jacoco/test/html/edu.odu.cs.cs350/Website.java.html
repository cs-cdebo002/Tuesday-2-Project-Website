    /**
     * For this project, it is necessary to convert any provided URLs into local directory paths
     * The path from the URL is stored into a temp variable and is resolved from the provided local path
     * The site root is identified and removed from the resolved path
     * Finally, the result temp is stored into path
     * 
     * @param
     * sPath - The path in the form of a string, using the getSPath getter
     * urLists- The list of URLs that's part of the website
     * @throws MalformedURLException
     * 
     * @output 
     * Resolved and shortened path
     */
    public String convertURLToPath(String sPath, List<URL> urLists) throws MalformedURLException {
        Path pathTemp = Paths.get(sPath);
        URL urlTemp = new URL("http://www.blank.com"); //URL constructor does not like empty strings
        ListIterator<URL> it = urLists.listIterator();

        while (it.hasNext()) {
            urlTemp = it.next();
            pathTemp = pathTemp.resolveSibling(urlTemp.getPath());
        }
            String update = pathTemp.toString();
            update = update.replace(sPath, "");

            return update;
        
    }
