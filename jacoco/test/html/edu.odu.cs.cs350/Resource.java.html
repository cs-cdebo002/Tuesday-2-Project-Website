package edu.odu.cs.cs350;

import java.util.*;
import java.net.URL;
import java.nio.file.Path;

enum Locality{ INTERPAGE, INTRAPAGE, EXTERNAL }
enum ResourceKind{ IMAGE, STYLESHEET, SCRIPT, ANCHOR, VIDEO, AUDIO, ARCHIVE, OTHER }

public class Resource{

    private Path path;
    private URL url;
    private List<HTMLDoc> foundOn;
    private Locality linkLocality;
    private long sizeOf;
    private ResourceKind resourceType;

    
    /**
     * Creates a new resource object using a url, alongside its other values
     * @param inUrl represents the resource's url
     * @param inLocality represents the resource's link locality (interpage, intrapage, or external)
     * @param inType represents the resource's type (image, stylesheet, script, anchor, video, audio, archive, or other)
     */
    public Resource(URL inUrl, Locality inLocality, ResourceKind inType){
        url = inUrl;
        linkLocality = inLocality;
        resourceType = inType;

        // Convert URL to path?
        this.setPath(null);
    }


    /**
     * Creates a new resource object using a path, alongside its other values
     * @param inPath represents the resource's path
     * @param inLocality represents the resource's link locality (interpage, intrapage, or external)
     * @param inType represents the resource's type (image, stylesheet, script, anchor, video, audio, archive, or other)
     */
    public Resource(Path inPath, Locality inLocality, ResourceKind inType){
        path = inPath;
        linkLocality = inLocality;
        resourceType = inType;

        // Convert path to URL?
        this.setUrl(null);
    }

    
    /**
     * Returns the path of the resource
     * @param n/a
     */
    public Path getPath(){ return path; }

    /**
     * Returns the url of the resource
     * @param n/a
     */
    public URL getUrl(){ return url; }

    /**
     * Returns the link locality
     * @param n/a
     */
    public Locality getLinkLocality(){ return linkLocality; }

    /**
     * Returns the size of the resource
     * @param n/a
     */
    public long getSizeOf(){ return sizeOf; }

    /**
     * Returns a vector containing every htmldoc the resource is found on
     * @param n/a
     */
    public List<HTMLDoc> getFoundOn() { return foundOn; }

    /**
     * Returns the resource's type
     * @param n/a
     */
    public ResourceKind getResourceKind(){ return resourceType; }


    
    /**
     * Sets the path of the resource to the provided string
     * @param inPath the provided string which is used to set the new path
     */
    public void setPath(Path inPath){ path = inPath; }

    /**
     * Sets the url of the resource to the provided string
     * @param inUrl the provided string which is used to set the new url
     */
    public void setUrl(URL inUrl){ url = inUrl; }

    /**
     * Sets the locality of the resource to the provided value
     * @param inLocality the value which is used to set the new locality
     */
    public void setLocality(Locality inLocality){ linkLocality = inLocality; }

    /**
     * Sets the size of the resource to the provided value
     * @param inSize the provided value which is used to set the new size
     */
    public void setSize(long inSize){ sizeOf = inSize; }

}
