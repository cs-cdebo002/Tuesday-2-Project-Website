package edu.odu.cs.cs350;

import java.io.IOException;
import java.net.HttpURLConnection;
import java.net.URL;


public class BrokenLinks {
    // This functiuon check to see if the link is broke by making a request to the link and seeing if the responsecode is 400 or higher.
    public static boolean isLinkBroken(String url) {
        try {
            URL link = new URL(url);
            HttpURLConnection connection = (HttpURLConnection) link.openConnection();
            connection.setRequestMethod("HEAD");
            int responseCode = connection.getResponseCode();
            return (responseCode >= 400);
        } catch (IOException e) {
            
            return true;
        }
    }

    public static void main(String[] args) {
        String url = "https://google.com";
        boolean isBroken = isLinkBroken(url);
        if (isBroken) {
            System.out.println("The link is broken.");
        } else {
            System.out.println("The link is valid.");
        }
    } 

}
