package edu.odu.cs.cs350;

import java.io.IOException;
import java.util.List;
import java.nio.file.Path;

public class ReportWriter {

    private String BaseFileName;
    private Website site;

    /**
     * Default constructor should never be used. "Disable" it by making it
     * private.
     */
    public ReportWriter()
    {
        
    }

    /**
     * Constructor with parameters
     *
     * @param website      The website object
     * @param BaseFileName The base file name
     */
    public ReportWriter(Website website, String BaseFileName)
    {
        this.site = website;
        this.BaseFileName = BaseFileName;
    }

    /**
     * Sets the source data (Website object) for the report writer.
     *
     * @param sourceData The website object.
     */
    public void setSourceData(Website sourceData)
    {
        this.site = sourceData;
    }

    /**
     * 
     * @return 
     */
    public Website getSourceData() {
        return this.site;
    }
    
    /**
     * Sets the base file name for the report writer.
     *
     * @param newBaseFileName The new base file name to set.
     */
    public void setBaseFileName(String newBaseFileName) {
        this.BaseFileName = newBaseFileName;
    }

    /**
     * Writes the report using the provided website data.
     *
     * @throws IOException
     */
    public void write() throws IOException {
        // Check if the website object or path is invalid
        if (site == null || site.getPath() == null) {
            System.out.println("Invalid website path or URL.");
            return;
        }

        TextReportWriter textFile = new TextReportWriter(site, BaseFileName);
        WebsiteBuilder websiteBuilder = new WebsiteBuilder();
        List<Path> files = websiteBuilder.walkDirectory(site.getPath());

        // Check if no files were found in the website path
        if (files.isEmpty()) {
            System.out.println("No files found in the website path.");
            return;
        }

        System.out.println("Website Path: " + site.getPath().toString());

        // Prepare the text report using the list of files and the base file name
        textFile.prepare(files, BaseFileName);
    }
}
