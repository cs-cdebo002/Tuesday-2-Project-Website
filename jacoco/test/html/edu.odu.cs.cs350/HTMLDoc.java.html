package edu.odu.cs.cs350;

import java.util.*;
import java.net.URL;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.net.MalformedURLException;
import java.io.*;
import java.beans.Transient;

public class HTMLDoc{

    private List<Resource> scriptList;
    private List<Resource> stylesheetList;
    private List<Resource> imageList;
    private List<Resource> anchorList;
    private String localPath;
    private List<URL> urlList;


    /**
     * Creates a new HTMLDoc object using a series of provided resource objects
     * @param inScripts a list of Resource objects representing scripts
     * @param inStylesheets a list of Resource objects representing stylesheets
     * @param inImages a list of Resource objects representing images
     * @param inAnchors a list of Resource objects representing anchors
     * @param inPath a string representing the local path of the HTML document
     */
    public HTMLDoc(List<Resource> inScripts, List<Resource> inStylesheets, List<Resource> inImages, List<Resource> inAnchors, String inPath, List<URL> inURLs){
        scriptList = inScripts;
        stylesheetList = inStylesheets;
        imageList = inImages;
        anchorList = inAnchors;
        localPath = inPath;
        urlList = inURLs;
    }



    /**
     * Creates a new HTMLDoc object using a series of provided resource objects (without a path)
     * @param inScripts a list of Resource objects representing scripts
     * @param inStylesheets a list of Resource objects representing stylesheets
     * @param inImages a list of Resource objects representing images
     * @param inAnchors a list of Resource objects representing anchors
     */
    public HTMLDoc(List<Resource> inScripts, List<Resource> inStylesheets, List<Resource> inImages, List<Resource> inAnchors){
        scriptList = inScripts;
        stylesheetList = inStylesheets;
        imageList = inImages;
        anchorList = inAnchors;
    }

    
    // Accessor functions

    /**
     * Returns the local path of the HTML document
     * @param n/a
     */
    public String getLocalPath(){return localPath;}

    /**
     * Returns the list of scripts used by the HTML document
     * @param n/a
     */
    public List<Resource> getScriptList(){ return scriptList; }
    
    /**
     * Returns the list of stylesheets used by the HTML document
     * @param n/a
     */
    public List<Resource> getStylesheetList(){ return stylesheetList; }
    
    /**
     * Returns the list of images used by the HTML document
     * @param n/a
     */
    public List<Resource> getImageList(){ return imageList; }
    
    /**
     * Returns the list of anchors used by the HTML document
     * @param n/a
     */
    public List<Resource> getAnchorList(){ return anchorList; }


    // Mutator functions

    /**
     * Adds a script resource to the html document's list of scripts
     * @param toAdd a resource object to be added to the list
     */
    public void addScript(Resource toAdd){ scriptList.add(toAdd); }
    
    /**
     * Adds a stylesheet resource to the html document's list of stylesheets
     * @param toAdd a resource object to be added to the list
     */
    public void addStylesheet(Resource toAdd){ stylesheetList.add(toAdd); }
    
    /**
     * Adds an image resource to the html document's list of images
     * @param toAdd a resource object to be added to the list
     */
    public void addImage(Resource toAdd){ imageList.add(toAdd); }
    
    /**
     * Adds an anchor resource to the html document's list of anchors
     * @param toAdd a resource object to be added to the list
     */
    public void addAnchor(Resource toAdd){ anchorList.add(toAdd); }

}
